46void Planer::step() {
47 // To start the frame, update the system state object. Each subsystem can
48 // take a read-only reference to the state object. This allows it to get the
49 // data it needs to perform its function without making unexpected changes
50 // to another subsystem's data.
51
52 updateState();
53
54 // Get user input in the form of a command. The command represents the
55 // user's request, and can be passed to other subsystems for execution.
56
57 Command command = input.doInput(state);
58
59 if (command.type == CommandType::SetMode) {
60 setMode(command.args.mode);
61 }
62
63 // Request the remaining systems do their processing for the frame, sending
64 // each of them the user command in case they need to execute it.
65 //
66 // !! Presently, we only pass the command to the motor subsystem. When we
67 // finish implementing memory and menu subsystems, we'll pass the command to
68 // them, too.
69
70 motor.doCommand(command);
71
72 display.step();
73}
